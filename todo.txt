Carbon footprint (kg CO2e) = Total grams of food wasted * Emission factor (kg CO2e/g)  4.5kg
const EMISSION_FACTOR = 4.5; // kg CO2e/kg of food

// Function to calculate carbon footprint
function calculateCarbonFootprint(initialWastedFood, reducedWastedFood) {
    // Calculate initial carbon footprint
    const initialCarbonFootprint = initialWastedFood * EMISSION_FACTOR;

    // Calculate reduced carbon footprint
    const reducedCarbonFootprint = reducedWastedFood * EMISSION_FACTOR;

    // Calculate carbon footprint reduction
    const carbonFootprintReduction = initialCarbonFootprint - reducedCarbonFootprint;

    return carbonFootprintReduction;
}

// Example usage
const initialWastedFood = 10; // kg
const reducedWastedFood = 5; // kg

const reduction = calculateCarbonFootprint(initialWastedFood, reducedWastedFood);
console.log(`Carbon Footprint Reduction: ${reduction} kg CO2e`);
